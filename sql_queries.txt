create database employee;


create table employee(id int,first_name varchar(255),last_name varchar(255),age int,mobile_number int,address varchar(255),primary key(id));

create table employee_salary(fk_$employee_id int,salary int,date date,primary key(fk_$employee_id ),foreign key(fk_$employee_id) references employee(id));

create table hobby(id int,name varchar(255),primary key(id));

create table employee_hobby(id int,fk_$employee_id int,fk_$hobby_id int,primary key(id),foreign key(fk_$employee_id) references employee(id),foreign key(fk_$hobby_id) references hobby(id));
